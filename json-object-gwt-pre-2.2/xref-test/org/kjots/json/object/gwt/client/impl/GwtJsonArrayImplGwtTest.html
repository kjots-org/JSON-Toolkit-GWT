115   /**
116    * Test the construction of a JSON array with a non-array-object argument.
117    * <p>
118    * This test asserts that the constructor of a JSON array will throw a
119    * {@link IllegalArgumentException} if the argument is not an array object.
120    */
121   public void testConstructionWithNonArrayObject() {
122     try {
123       new GwtJsonArrayImpl(JavaScriptObject.createObject());
124       
125       fail("Expected " + IllegalArgumentException.class.getName());
126     }
127     catch (IllegalArgumentException iae) {
128     }
